## SQL Example

```sql
-- FULL WORKFLOW REVIEW

-- Step 1: Create a backup before making changes
-- pg_dump mydb > backup_before_cleaning.sql

-- Step 2: Check current data quality
SELECT
    COUNT(*) AS total_records,
    COUNT(*) - COUNT(email) AS missing_emails,
    COUNT(*) - COUNT(phone) AS missing_phones
FROM customers;

-- Step 3: Clean the data
UPDATE customers SET email = LOWER(TRIM(email));
UPDATE customers SET phone = REPLACE(phone, '-', '');
UPDATE customers SET name = TRIM(name);
UPDATE customers SET city = INITCAP(LOWER(city));

-- Step 4: Remove confirmed duplicates (keep lowest id)
DELETE FROM customers
WHERE id NOT IN (
    SELECT MIN(id) FROM customers GROUP BY email
);
```


```sql
-- Step 5: Export cleaned data to CSV
-- COPY customers TO '/tmp/clean_customers.csv' CSV HEADER;

-- Step 6: Verify with post-cleaning counts
SELECT COUNT(*) AS cleaned_total,
       COUNT(DISTINCT email) AS unique_emails
FROM customers;

-- Step 7: Create backup after cleaning
-- pg_dump mydb > backup_after_cleaning.sql
```
