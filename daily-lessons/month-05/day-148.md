## SQL Example

```sql
-- Week 20 Review: Combined operations query
-- 1. Reporting: Monthly revenue with comparison
SELECT DATE_TRUNC('month', order_date) AS month,
       SUM(total) AS revenue,
       LAG(SUM(total)) OVER (ORDER BY DATE_TRUNC('month', order_date)) AS prev_month,
       ROUND(((SUM(total) - LAG(SUM(total)) OVER (ORDER BY DATE_TRUNC('month', order_date)))
       / NULLIF(LAG(SUM(total)) OVER (ORDER BY DATE_TRUNC('month', order_date)), 0)) * 100, 1) AS growth_pct
FROM orders
GROUP BY DATE_TRUNC('month', order_date)
ORDER BY month DESC LIMIT 12;

-- 2. API Pattern: Paginated search with total count
SELECT id, name, email, created_at,
       COUNT(*) OVER() AS total_count
FROM customers
WHERE name ILIKE '%search_term%'
ORDER BY created_at DESC
LIMIT 20 OFFSET 0;

-- 3. Health Check: Connection and performance summary
SELECT
  (SELECT COUNT(*) FROM pg_stat_activity WHERE state = 'active') AS active_queries,
  (SELECT COUNT(*) FROM pg_stat_activity WHERE state = 'idle') AS idle_connections,
  (SELECT pg_size_pretty(pg_database_size(current_database()))) AS db_size;
```
