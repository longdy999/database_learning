## SQL Example

```sql
-- Month 5 Comprehensive Review Query
-- Demonstrates: PostgreSQL features, reporting, security, and monitoring

-- 1. PostgreSQL-specific: JSONB data handling
SELECT id, profile->>'name' AS name,
       profile->>'email' AS email
FROM users
WHERE profile @> '{"role": "admin"}'::jsonb;

-- 2. Backup verification: Check last backup timestamp
SELECT backup_type, backup_date, size_mb, status
FROM backup_log
WHERE backup_date >= CURRENT_DATE - INTERVAL '7 days'
ORDER BY backup_date DESC;

-- 3. Security: Audit user permissions
SELECT grantee, table_name, privilege_type
FROM information_schema.table_privileges
WHERE table_schema = 'public'
ORDER BY grantee, table_name;

-- 4. Reporting: Complete business dashboard query
SELECT
  (SELECT COUNT(*) FROM orders WHERE order_date = CURRENT_DATE) AS today_orders,
  (SELECT SUM(total) FROM orders WHERE order_date = CURRENT_DATE) AS today_revenue,
  (SELECT COUNT(*) FROM customers WHERE created_at >= CURRENT_DATE) AS new_customers,
  (SELECT COUNT(*) FROM pg_stat_activity WHERE state = 'active') AS active_queries;
```
