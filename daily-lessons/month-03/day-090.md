## SQL Example
```sql
-- Restaurant Management System - Core Tables

CREATE TABLE staff_roles (
    role_id SERIAL PRIMARY KEY,
    role_name VARCHAR(30) UNIQUE NOT NULL
);

CREATE TABLE staff (
    staff_id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    role_id INT REFERENCES staff_roles(role_id),
    phone VARCHAR(20),
    hire_date DATE NOT NULL DEFAULT CURRENT_DATE,
    is_active BOOLEAN DEFAULT TRUE
);

CREATE TABLE menu_categories (
    category_id SERIAL PRIMARY KEY,
    category_name VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE menu_items (
    item_id SERIAL PRIMARY KEY,
    item_name VARCHAR(100) NOT NULL,
    category_id INT REFERENCES menu_categories(category_id),
    price DECIMAL(8,2) NOT NULL CHECK (price > 0),
    description TEXT,
    is_available BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```


```sql
-- Additional tables for the Restaurant system
CREATE TABLE restaurant_tables (
    table_id SERIAL PRIMARY KEY,
    table_number INT UNIQUE NOT NULL,
    capacity INT NOT NULL CHECK (capacity > 0),
    location VARCHAR(30) DEFAULT 'indoor'
);

CREATE TABLE customers (
    customer_id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50),
    phone VARCHAR(20) UNIQUE,
    email VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE reservations (
    reservation_id SERIAL PRIMARY KEY,
    customer_id INT REFERENCES customers(customer_id),
    table_id INT REFERENCES restaurant_tables(table_id),
    reservation_date DATE NOT NULL,
    reservation_time TIME NOT NULL,
    party_size INT NOT NULL CHECK (party_size > 0),
    status VARCHAR(20) DEFAULT 'confirmed'
);

CREATE TABLE orders (
    order_id SERIAL PRIMARY KEY,
    table_id INT REFERENCES restaurant_tables(table_id),
    staff_id INT REFERENCES staff(staff_id),
    customer_id INT REFERENCES customers(customer_id),
    order_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) DEFAULT 'open',
    total_amount DECIMAL(10,2)
);

CREATE TABLE order_items (
    order_item_id SERIAL PRIMARY KEY,
    order_id INT NOT NULL REFERENCES orders(order_id),
    item_id INT NOT NULL REFERENCES menu_items(item_id),
    quantity INT NOT NULL CHECK (quantity > 0),
    unit_price DECIMAL(8,2) NOT NULL,
    special_notes TEXT
);
```

## Reading Practice
Review your completed design against this professional checklist:
- Every table has a clear primary key
- All relationships use proper foreign keys
- Naming follows plural_lowercase_underscore convention
- Constraints protect data integrity (NOT NULL, CHECK, UNIQUE)
- Lookup tables used for categories and statuses
- Audit fields included where appropriate
- No repeating column groups
- Design supports real restaurant operations
