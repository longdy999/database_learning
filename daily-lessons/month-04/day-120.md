## SQL Example

```sql
-- ===== PART 1: Create 3 Business Views =====

-- View 1: Daily Sales Dashboard
CREATE VIEW v_daily_sales AS
SELECT order_date,
       COUNT(*) AS num_orders,
       SUM(total_amount) AS daily_revenue,
       AVG(total_amount) AS avg_order_value
FROM orders
WHERE status = 'completed'
GROUP BY order_date
ORDER BY order_date DESC;

-- View 2: Product Performance Report
CREATE VIEW v_product_performance AS
SELECT p.product_name, p.category,
       COUNT(o.order_id) AS times_ordered,
       SUM(o.quantity) AS total_units_sold,
       SUM(o.quantity * p.unit_price) AS total_revenue
FROM products p
LEFT JOIN orders o ON p.product_id = o.product_id
GROUP BY p.product_name, p.category;

-- View 3: Customer Lifetime Value
CREATE VIEW v_customer_value AS
SELECT c.customer_id, c.customer_name, c.email,
       COUNT(o.order_id) AS total_orders,
       COALESCE(SUM(o.total_amount), 0) AS lifetime_value,
       MAX(o.order_date) AS last_order_date
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.customer_id, c.customer_name, c.email;


-- ===== PART 2: Add Indexes for 5 Most Common Queries =====

-- Before indexes: run EXPLAIN to see Seq Scans
EXPLAIN ANALYZE SELECT * FROM orders WHERE customer_id = 42;
EXPLAIN ANALYZE SELECT * FROM orders WHERE order_date >= '2024-01-01';

-- Create indexes
CREATE INDEX idx_orders_customer_id ON orders(customer_id);
CREATE INDEX idx_orders_date ON orders(order_date);
CREATE INDEX idx_orders_status ON orders(status);
CREATE INDEX idx_products_category ON products(category);
CREATE INDEX idx_orders_product_id ON orders(product_id);

-- After indexes: run EXPLAIN to confirm Index Scans
EXPLAIN ANALYZE SELECT * FROM orders WHERE customer_id = 42;
EXPLAIN ANALYZE SELECT * FROM orders WHERE order_date >= '2024-01-01';

-- ===== PART 3: Transaction for Order Processing =====

CREATE OR REPLACE FUNCTION process_full_order(
    p_customer_id INT,
    p_product_id INT,
    p_quantity INT,
    p_payment_method TEXT
)
RETURNS TEXT AS $$
DECLARE
    v_stock INT;
    v_price NUMERIC;
    v_order_id INT;
    v_total NUMERIC;
BEGIN
    -- Step 1: Check inventory
    SELECT stock_quantity, unit_price INTO v_stock, v_price
    FROM products WHERE product_id = p_product_id FOR UPDATE;

    IF v_stock < p_quantity THEN
        RAISE EXCEPTION 'Insufficient stock';
    END IF;

    v_total := v_price * p_quantity;

    -- Step 2: Deduct inventory
    UPDATE products SET stock_quantity = stock_quantity - p_quantity
    WHERE product_id = p_product_id;

    -- Step 3: Create order
    INSERT INTO orders (customer_id, product_id, quantity, total_amount, status, order_date)
    VALUES (p_customer_id, p_product_id, p_quantity, v_total, 'confirmed', CURRENT_DATE)
    RETURNING order_id INTO v_order_id;

    -- Step 4: Record payment
    INSERT INTO payments (order_id, amount, method, status, paid_at)
    VALUES (v_order_id, v_total, p_payment_method, 'completed', NOW());

    RETURN 'Order #' || v_order_id || ' processed. Total: $' || v_total;
EXCEPTION
    WHEN OTHERS THEN
        RETURN 'FAILED: ' || SQLERRM;
END;
$$ LANGUAGE plpgsql;


-- ===== PART 4: CTE for Monthly Trend Analysis =====

WITH monthly_revenue AS (
    SELECT DATE_TRUNC('month', order_date) AS month,
           SUM(total_amount) AS revenue,
           COUNT(*) AS order_count
    FROM orders
    WHERE status = 'completed'
    GROUP BY DATE_TRUNC('month', order_date)
),
growth AS (
    SELECT month, revenue, order_count,
           LAG(revenue) OVER (ORDER BY month) AS prev_revenue,
           ROUND(((revenue - LAG(revenue) OVER (ORDER BY month))
                  / LAG(revenue) OVER (ORDER BY month)) * 100, 2) AS growth_pct
    FROM monthly_revenue
)
SELECT month, revenue, order_count, prev_revenue, growth_pct
FROM growth ORDER BY month;
```

## Technical Vocabulary

| Term | Definition |
|------|-----------|
| Business View | A view designed for non-technical report consumers |
| FOR UPDATE | Locks selected rows during a transaction |
| Index Scan vs Seq Scan | Fast indexed lookup vs slow full table scan |
| Trend Analysis | Comparing metrics over time periods |
| LAG() | Window function accessing previous row value |
| Growth Percentage | Rate of change between periods |
| COALESCE | Returns first non-null value from a list |
| DATE_TRUNC | Truncates a timestamp to specified precision |
| Lifetime Value | Total revenue from a single customer |
| Schema Design | Planning tables and relationships for a system |
