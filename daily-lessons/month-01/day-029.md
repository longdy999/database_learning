## SQL Example
```sql
-- COMPLETE MONTH 1 REVIEW: Build from memory
CREATE TABLE employees (
    id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE,
    department VARCHAR(50),
    salary DECIMAL(10,2),
    hire_date DATE
);

INSERT INTO employees (id, name, email, department, salary, hire_date) VALUES
(1, 'Sokha', 'sokha@co.com', 'Engineering', 2500.00, '2021-03-15'),
(2, 'Dara', 'dara@co.com', 'Sales', 1800.00, '2022-07-01'),
(3, 'Chantha', NULL, 'Engineering', 2200.00, '2020-01-10'),
(4, 'Visal', 'visal@co.com', 'Marketing', 1900.00, '2023-06-20');

SELECT name, salary FROM employees
WHERE department IN ('Engineering', 'Sales') AND salary BETWEEN 2000 AND 3000 AND email IS NOT NULL;

UPDATE employees SET salary = salary * 1.05 WHERE department = 'Sales' AND hire_date < '2022-01-01';
DELETE FROM employees WHERE department = 'Marketing' AND salary < 1500;
```
