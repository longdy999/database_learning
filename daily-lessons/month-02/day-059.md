## SQL Example

### Month 2 Skill Checklist Queries

```sql
-- 1. ORDER BY with multiple columns
SELECT * FROM products
ORDER BY category ASC, price DESC;

-- 2. LIMIT and OFFSET
SELECT * FROM employees ORDER BY hire_date DESC LIMIT 10 OFFSET 20;

-- 3. LIKE pattern matching
SELECT * FROM customers WHERE email LIKE '%@gmail.com';

-- 4. DISTINCT
SELECT DISTINCT city, country FROM suppliers;

-- 5. INNER JOIN
SELECT e.name, d.dept_name
FROM employees e
INNER JOIN departments d ON e.dept_id = d.dept_id;
```


```sql
-- 6. LEFT JOIN (keep all from left table)
SELECT c.name, o.order_id
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id;

-- 7. Aggregate functions
SELECT
    COUNT(*) AS total_orders,
    SUM(amount) AS total_revenue,
    AVG(amount) AS avg_order,
    MIN(amount) AS smallest_order,
    MAX(amount) AS largest_order
FROM orders;

-- 8. GROUP BY with HAVING
SELECT category, COUNT(*) AS product_count, AVG(price) AS avg_price
FROM products
GROUP BY category
HAVING COUNT(*) >= 5 AND AVG(price) > 20;

-- 9. Subquery in WHERE
SELECT name, salary
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);

-- 10. Complex multi-table query
SELECT
    c.customer_name,
    COUNT(DISTINCT o.order_id) AS order_count,
    SUM(oi.quantity * oi.unit_price) AS total_spent
FROM customers c
JOIN orders o ON c.customer_id = o.customer_id
JOIN order_items oi ON o.order_id = oi.order_id
WHERE o.order_date >= '2024-01-01'
GROUP BY c.customer_name
HAVING SUM(oi.quantity * oi.unit_price) > 500
ORDER BY total_spent DESC
LIMIT 10;
```
