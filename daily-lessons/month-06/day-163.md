## SQL Example

```sql
-- Query 1: Top 5 Best-Selling Books
SELECT b.title, b.author, SUM(oi.quantity) AS total_sold
FROM books b
JOIN order_items oi ON b.book_id = oi.book_id
JOIN orders o ON oi.order_id = o.order_id
WHERE o.status != 'cancelled'
GROUP BY b.title, b.author
ORDER BY total_sold DESC
LIMIT 5;

-- Query 2: Monthly Revenue Report
SELECT DATE_TRUNC('month', order_date) AS month,
       COUNT(*) AS total_orders,
       SUM(total_amount) AS revenue
FROM orders
WHERE status IN ('delivered', 'shipped', 'confirmed')
GROUP BY DATE_TRUNC('month', order_date)
ORDER BY month;

-- Query 3: Customer Lifetime Value
SELECT c.first_name, c.last_name,
       COUNT(o.order_id) AS order_count,
       SUM(o.total_amount) AS lifetime_value
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
    AND o.status != 'cancelled'
GROUP BY c.customer_id, c.first_name, c.last_name
ORDER BY lifetime_value DESC NULLS LAST;

-- Query 4: Average Rating by Category
SELECT cat.category_name,
       ROUND(AVG(r.rating), 2) AS avg_rating,
       COUNT(r.review_id) AS review_count
FROM categories cat
JOIN books b ON cat.category_id = b.category_id
JOIN reviews r ON b.book_id = r.book_id
GROUP BY cat.category_name
ORDER BY avg_rating DESC;
```


```sql
-- Query 5: Low Stock Alert (books with less than 20 in stock)
SELECT title, author, stock_quantity, price
FROM books
WHERE stock_quantity < 20
ORDER BY stock_quantity ASC;

-- Query 6: Customers Who Never Ordered
SELECT c.first_name, c.last_name, c.email
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
WHERE o.order_id IS NULL;

-- Query 7: Order Status Distribution
SELECT status, COUNT(*) AS order_count,
       ROUND(COUNT(*)::DECIMAL / SUM(COUNT(*)) OVER() * 100, 1) AS percentage
FROM orders
GROUP BY status
ORDER BY order_count DESC;

-- Query 8: Books Never Ordered
SELECT b.title, b.author, b.price
FROM books b
WHERE b.book_id NOT IN (
    SELECT DISTINCT book_id FROM order_items
);

-- Query 9: Revenue by Category
SELECT cat.category_name, SUM(oi.quantity * oi.unit_price) AS category_revenue
FROM categories cat
JOIN books b ON cat.category_id = b.category_id
JOIN order_items oi ON b.book_id = oi.book_id
JOIN orders o ON oi.order_id = o.order_id
WHERE o.status != 'cancelled'
GROUP BY cat.category_name
ORDER BY category_revenue DESC;

-- Query 10: Recent Orders with Customer Details
SELECT o.order_id, c.first_name || ' ' || c.last_name AS customer,
       o.order_date, o.status, o.total_amount
FROM orders o
JOIN customers c ON o.customer_id = c.customer_id
ORDER BY o.order_date DESC
LIMIT 10;
```

## Technical Vocabulary
- **Lifetime value**: Total revenue a single customer generates over time
- **Window function**: Calculation across rows without collapsing groups
- **DATE_TRUNC**: Rounds timestamps to specified precision (month, year)
- **NULLS LAST**: Places NULL values at end of sorted results
